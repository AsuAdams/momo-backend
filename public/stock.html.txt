<!DOCTYPE html>
<html>
<head>
    <title>Buy Stock</title>
</head>
<body>
    <h1>Buy This Share</h1>
    <button id="buyBtn">Buy Now</button>
    <p id="status"></p>

    <script>
        document.getElementById("buyBtn").addEventListener("click", async () => {
            const username = localStorage.getItem("username");
            if (!username) {
                document.getElementById("status").innerText = "Please sign up first!";
                return;
            }

            const res = await fetch("/buy", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ username, amount: 1000, currency: "UGX" })
            });

            const data = await res.json();
            if (data.success) {
                document.getElementById("status").innerText = "✅ Payment request sent! Please approve on your phone.";
            } else {
                document.getElementById("status").innerText = "❌ " + data.message;
            }
        });
    </script>
</body>
</html>
